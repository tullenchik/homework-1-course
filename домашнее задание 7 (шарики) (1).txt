//Bp==Из урны с 10 пронумерованными шариками вынимают по одному шарику.Подсчитать общее количество ситуаций, когда номер хотя бы одного вынутого шарика совпадает с порядковым номеров действия "вынимания", например, когда шарик №3 юудет вынут 3-им по порядку
#include <iostream>//для работы с консолью
#include <math.h>//для работы с математическими действиями
using namespace std;//пространство имен стд
//функция расчет факториала
long long fact(int f) 
{
    if (f < 0)
    {
        return 0;
    }
    if (f == 0)
    {
        return 1;
    }
    else
    {
        return f * fact(f - 1);
    }
}
int main() 
{
    setlocale(LC_ALL, "rus");//русифкация
    //объявление переменных
    int informacia=10;
    long long answer;
    answer= fact(informacia) - (fact(informacia) / exp(1.0));
    cout << "Шариков 10, следовательно общее количество ситуаций, удовлетворяющее условию:";
    cout << "\nСитуаций: " << answer;
    return 0;
}
