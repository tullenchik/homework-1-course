		cout << "Неправильный ввод";
		return 0;
	}

		cout << "Неправильный ввод";
		return 0;
	}
	cout << "Сумма введенных чисел: " << a + b << endl;//сумма
	cout << "Если из первого числа вычитать второе, то разность чисел: " << a - b << endl;//разность
	cout<<". А Если из второго числа вычитать первое, то разность чисел: " << b - a << endl;//разность2
	cout << "Произведение чисел: " << a * b << endl;//умножение
	cout << "Если делить первое число на второе, то  " << endl;//деление+ОДЗ
	if (b != 0) {
		cout << "результат частного чисел равен:" << a/b << endl;
	} else {
		cout << "результат невозможен";
	}
	cout << "Если делить второе число на первое, то " << endl;
	if (a != 0) {
		cout << "результат частного чисел равен:" << b/a << endl;
	} else {
		cout << "результат невозможен" << endl;
	}
	return 0;
}

номер 4 ЕЩЕ УРАВНЕНИЕ

#include <iostream>//для вывода в консоль
#include <cstdlib>
#include <cmath>//для квадратного корня
using namespace std;

int main()
{
	setlocale(0, "");//русификация консоли
	double a, b, c, d;//d-дискриминант
	cout << "Введите переменную a для квадратного уравнения" << endl;//ввод 3 переменных
	cin >> a;
	if (!cin) {//проверка на дурака
		cout << "Неправильный ввод" << endl;
		return 0;
	}
	cout << "Введите переменную b" << endl;
	cin >> b;
	if (!cin) {//проверка на дурака
		cout << "Неправильный ввод" << endl;
		return 0;
	}
	cout << "Введите переменную с" << endl;
	cin >> c;
	if (!cin) {//проверка на дурака
		cout << "Неправильный ввод" << endl;
		return 0;
	}
	if ((a == 0) && (b == 0) && (c == 0)) {//первый случай
		cout << "Любое решение" << endl;
		return 0;
	}
	if ((a == 0) && (b == 0) && (c != 0)) {//второй случай
		cout << "Решений нет" << endl;
		return 0;
	}
	if ((a == 0) && (b != 0) && (c != 0)) {//третий случай
		cout << "Решение уравнения: " << -1 * c / b << endl;
		return 0;
	}
	if ((a != 0) && (b != 0) && (c == 0)) {//четвертый случай
		cout << "Решение уравнения: 0, " << -1 * b / a << endl;
		return 0;
	}
	if ((a != 0) && (b == 0) && (c == 0)) {//пятый случай
		cout << "Решение уравнения 0" << endl;
		return 0;
	}
	if (((a != 0) && (b == 0) && (c != 0)) || ((a == 0) && (b != 0) && (c == 0))) {//седьмой и восьмой случай
		if (((a > 0) && (c > 0)) || ((a < 0) && (c < 0))) {
			cout << "Решений нет" << endl;
			return 0;
		}
		if (((a > 0) && (c < 0)) || ((a < 0) && (c > 0))) {
			cout << "Первый корень" << sqrt(-1 * c / a) << endl;
			cout << "Второй корень" << -1 * sqrt(-1 * c / a) << endl;
			return 0;
		}
	}
	d = b * b - 4 * a * c;//вычисления дискриминанта
	if (d < 0) {
		cout << "Решений нет" << endl;
		return 0;
	}
	if (d >= 0) {
		cout << "Первый корень уравнения: " << (-1 * b + sqrt(d)) / 2 * a << endl;//вычисления 2 корней
		cout << "Второй корень уравнения: " << (-1 * b - sqrt(d)) / 2 * a << endl;
		return 0;
	}
	return 0;
}

#include <iostream>//библиотека для вывода в консоли
using namespace std;//пространство имен стд
#include <string>//для переменных string

int main()
{
	setlocale(0, "");//русификация консоли
	bool t, r;//логические переменные
	cout << "Посмотрите в окно. Если шторы закрыты, то напишите 0. Если шторы открыты, то напишите 1, если светло на улице, или 0, если темно." << endl;
	cin >> t;//понимаем закрыты ли шторы или нет
	if (!cin) {//проверка на дурака
		cout << "Ввод неверен" << endl;
		return 0;
	}
	if ((t != 0) && (t != 1)) {
		cout << " Ввод неверен" << endl;
		return 0;
	}
	cout << "Включена ли лампа? Если да, напишите 1, если нет - 0." << endl;
	cin>> r;//понимаем влючена ли лампа
	if (!cin) {//проверка на дурака
		cout << "Ввод неверен" << endl;
		return 0;
	}
	if ((r != 0) && (r != 1)) {
		cout << " Ввод неверен" << endl;
		return 0;
	}
	if ((t == 0) && (r == 0)) {//первый случай (лампа не горит, а на улице темно или закрыты шторы)
		cout << "В комнате темно" << endl;
		return 0;
	}
	else {//во всех остальных случаях лиюо лампа включена, либо на улице светло
		cout << "В комнате светло" << endl;
		return 0;
	}
	return 0;
}
