Номер 1 КОНУС

#include <iostream>//библиотека для вывода в консоль
#include <cstdlib>
#include <cmath>//для квадратного корня
using namespace std;//
int main() 
{
	setlocale(0, "");//
	double h,r1,r2,l,pi=3.1415926535;//
	cout << "Введите высоту усеченного конуса h: " << endl;
	cin >> h;//ввод переменных
	if (!cin) {//проверка на дурака
		cout << "Ввод неверен" << endl;
		return 0;
	}
	cout << "Введите радиус нижнего основания r1: " << endl;
	cin >> r1;
	if (!cin) {//проверка на дурака
		cout << "Ввод неверен" << endl;
		return 0;
	}
	cout << "Введите радиус верхнего основания r2: " << endl;
		cin >> r2;
	if (!cin) {//проверка на дурака
		cout << "Ввод неверен" << endl;
		return 0;
	}
	if ((h <= 0) || (r1 <= 0) || (r2 <= 0)) {//проверка на дурака
		cout << "Ввод неверен" << endl;
		return 0;
	}
	l = sqrt(h * h + fabs(r1 - r2) * fabs(r1 - r2));//вычисление
	cout << "Объем усеченного конуса: " << (pi*h*(r1 * r1 + r1 * r2 + r2 * r2))/3.0 << endl;
	cout << "Полная поверхность усеченного конуса: " << pi*(r1 * r1 + (r1 + r2) * l + r2 * r2) << endl;
}

Номер 2 Разветвление

#include <iostream>//библиотека для вывода в консоль
#include <cstdlib>
#include <cmath>//библиотека для квадратного корня и модуля
using namespace std;//простарнство имен стд

int main()
{
	setlocale(0, "");//русификация консоли
	double a, x;//ввод переменных
	cout << "Введите произвольную а: " << endl;
	cin >> a;
	if (!cin) {//проверка на дурака
		cout << "Ввод неверен" << endl;
		return 0;
	}
	cout << "Введите произвольную x: " << endl;
	cin >> x;
	if (!cin) {//проверка на дурака
		cout << "Ввод неверен" << endl;
		return 0;
	}
	if (fabs(x) < 1) {//ОДЗ
		if (x == 0) {
			cout << "Не соблюдено ОДЗ натурального логарифма. Результат невозможен" << endl;
			return 0;
		}
		else {
			cout << "Результат: " << a * log(fabs(x));//вычисление
			return 0;
		}
	}
	if (fabs(x) >= 1) {//ОДЗ
		if ((a - x * x) < 0) {
			cout << "Не соблюдено одз квадратного корня. Результат невозможен";
			return 0;
		}
		else {
			cout << "Результат: " << sqrt(a - x * x) << endl;//вычисление
			return 0;
		}
	}
}

Номер 4 ПОРЯДОК

#include <iostream>//библиотека для консоли
using namespace std;//пространство имен

int main()
{
    setlocale(0, "");
    int* arr; // указатель для выделения памяти под массив
    int size; // размер массива
    size = 10;
    arr = new int[size]; // выделение памяти под массив
    cout << "Заполните массив: " << endl;
    for (int i = 0; i < size; i++) {//заполнение массива
        cout << "arr[" << i << "] = ";//i-номер элемента массива
        cin >> arr[i];
        if (!cin) {
            cout << "Ввод неверен" << endl;
            return 0;
        }
    }
    int obmen; // временная переменная 
    for (int i = 0; i < size - 1; i++) {//сортировка
        for (int j = 0; j < size - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                // меняем элементы местами
                obmen = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = obmen;
            }
        }
    }
    cout << "Отсортированный массив: " << endl;
    for (int i = 0; i < size; i++) {//выведем массив
        cout << arr[i] << " ";
    }
    cout << endl;
    return 0;
}
